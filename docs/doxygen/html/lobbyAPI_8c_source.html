144Error_Et lobby_gameLoop(LobbyGame_St* const game) {
145 if (game == NULL) return ERROR_NULL_POINTER;
146
147 f32 dt = GetFrameTime();
148 static f32 lobbyTextXPos;
149
150 updatePlayer(&game->player, platforms, platformCount, dt);
151 game->cam.target = game->player.position;
152
153 toggleSkinMenu(game);
154
155 if (game->playerVisuals.isTextureMenuOpen) {
156 choosePlayerTexture(&game->player, game);
157 }
158
159 // Collision check with game zone
160 for (u8 i = 1; i < __gameSceneCount; ++i) {
161 if (CheckCollisionCircleRec(game->player.position, game->player.radius, game->subGameManager.gameHitboxes[i])) {
162 if (!game->subGameManager.gameHitGracePeriodActive) {
163 game->subGameManager.currentScene = i;
164 game->subGameManager.needGameInit = true;
165 game->subGameManager.gameHitGracePeriodActive = true;
166 }
167 } else if (game->subGameManager.gameHitGracePeriodActive) {
168 game->subGameManager.gameHitGracePeriodActive = false;
169 }
170 }
171
172 BeginDrawing(); {
173 ClearBackground(RAYWHITE);
174
175 BeginMode2D(game->cam); {
176 DrawCircle(0, 0, 10, RED); // Debug origin marker
177 drawPlayer(game, &game->player);
178 drawPlatforms(platforms, platformCount);
179
180 for (u8 i = 1; i < __gameSceneCount; ++i) {
181 DrawRectangleRec(game->subGameManager.gameHitboxes[i], RED); // Debug hitbox
182 }
183 } EndMode2D();
184
185 lobbyTextXPos = (systemSettings.video.width - MeasureText("Multi-Mini-Games", 20)) / 2.0f;
186 DrawText("Multi-Mini-Games", lobbyTextXPos, 20, 20, PURPLE);
187
188 drawSkinButton();
189
190 if (game->playerVisuals.isTextureMenuOpen) {
191 drawMenuTextures(game);
192 }
193 } EndDrawing();
194
195 return OK;
196}

198Error_Et lobby_freeGame(LobbyGame_St** game) {
199 if (game == NULL || *game == NULL) return ERROR_NULL_POINTER;
200 LobbyGame_St* gameRef = *game;
201
202 for (u8 i = 1; i < __gameSceneCount; ++i) {
203 if (gameRef->subGameManager.miniGames[i] == NULL) continue;
204 gameRef->subGameManager.miniGames[i]->freeGame(&gameRef->subGameManager.miniGames[i]);
205 gameRef->subGameManager.miniGames[i] = NULL;
206 }
207
208 for (u32 i = 1; i < __playerTextureCount; ++i) {
209 UnloadTexture(gameRef->playerVisuals.textures[i]);
210 }
211
212 UnloadTexture(logoSkinButton);
213
214 free(gameRef);
215 *game = NULL;
216
217 CloseWindow();
218
219 return OK;
