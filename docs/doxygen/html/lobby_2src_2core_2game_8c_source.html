136void resolveCircleRectCollision(Player_st* player, Rectangle rect) {
137 // Search the position that is closest to the circle on the rectangle
138 f32 closestX = Clamp(player->position.x, rect.x, rect.x + rect.width);
139 f32 closestY = Clamp(player->position.y, rect.y, rect.y + rect.height);
140
141 f32 dx = player->position.x - closestX;
142 f32 dy = player->position.y - closestY;
143
144 f32 distSq = dx*dx + dy*dy;
145 f32 r = player->radius;
146
147 if (distSq >= r * r)
148 return;
149
150 f32 dist = sqrtf(distSq);
151 if (dist == 0)
152 return;
153
154 f32 penetration = r - dist;
155
156 f32 nx = dx / dist;
157 f32 ny = dy / dist;
158
159 // Position correction
160 player->position.x += nx * penetration;
161 player->position.y += ny * penetration;
162
163 // Speed resolution along the dominant axis
164 if (fabsf(nx) > fabsf(ny)) {
165 player->velocity.x = 0;
166 }
167 else {
168 player->velocity.y = 0;
169
170 // Ground
171 if (ny < 0) {
172 player->onGround = true;
173 player->nbJumps = 0;
174 player->coyoteTimer = COYOTE_TIME;
175 }
176 }
177}
