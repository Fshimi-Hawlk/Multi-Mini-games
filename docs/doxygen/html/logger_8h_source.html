351void print_stack_trace(unsigned int targetDepth) {
352 /* Print stack trace until main */
353 fprintf(stderr, "%sStack trace\033[0m:\n", getLevelColor(LOGGING_LEVEL_TRACE));
354 if (log_file) {
355 fprintf(log_file, "Stack trace:\n");
356 }
357
358 targetDepth = (targetDepth + startDepth) > MAX_TRACE_BACK_FRAMES ? MAX_TRACE_BACK_FRAMES : targetDepth + startDepth;
359
360 for (unsigned int depth = startDepth; depth < targetDepth; depth++) {
361 char buf[512];
362 if (get_caller_info(buf, sizeof(buf), depth) != 0) {
363 fprintf(stderr, " Error getting caller info at depth %u\n", depth);
364 if (log_file) {
365 fprintf(log_file, " Error getting caller info at depth %u\n", depth);
366 }
367 break;
368 }
369
370 /* Parse buffer line by line */
371 char *line = buf;
372 char func_name[256] = {0};
373 int found_func = 0;
374
375 while (line && *line) {
376 char *next_line = strchr(line, '\n');
377 if (next_line) *next_line = '\0';
378
379 /* Extract function name for main detection */
380 if (!found_func && strncmp(line, "Caller function: ", 17) == 0) {
381 char *func_start = line + 17;
382 char *func_end = strchr(func_start, '\n');
383
384 if (!func_end) {
385 func_end = func_start + strlen(func_start);
386 }
387
388 size_t func_len = func_end - func_start;
389 if (func_len >= sizeof(func_name)) {
390 func_len = sizeof(func_name) - 1;
391 }
392
393 strncpy(func_name, func_start, func_len);
394 func_name[func_len] = '\0';
395 found_func = 1;
396 }
397
398 /* Print to stderr (colored) */
399 if (strncmp(line, "Module: ", 8) == 0) {
400 fprintf(stderr, " %s\n", line);
401
402 } else if (strncmp(line, "Caller function: ", 17) == 0) {
403 char *func = line + 17;
404 fprintf(stderr, " %sCaller function: %s%s\033[0m\n", getLevelColor(LOGGING_LEVEL_TRACE), functionColor, func);
405
406 } else if (strncmp(line, " at ", 5) == 0) {
407 char *file_line = line + 5;
408 char *colon = strrchr(file_line, ':');
409 if (colon) {
410 *colon = '\0';
411 const char *file = file_line;
412 const char *line_num = colon + 1;
413 fprintf(stderr, " %sat %s%s:%s%s\033[0m\n", getLevelColor(LOGGING_LEVEL_TRACE), fileColor, file, lineColor, line_num);
414
415 } else {
416 fprintf(stderr, " %s%s\033[0m\n", yellow, line);
417 }
418
419 }
420 // else {
421 // fprintf(stderr, " %s\n", line);
422 // }
423
424 /* Print to log file (plain text) */
425 if (log_file) {
426 fprintf(log_file, " %s\n", line);
427 }
428
429 line = next_line ? next_line + 1 : NULL;
430 if (next_line) *next_line = '\n';
431 }
432
433 /* Stop if we reach the main function */
434 if (func_name[0] && strcmp(func_name, "main") == 0) {
435 break;
436 }
437 }
438
439 if (log_file) {
440 fflush(log_file);
441 }
442}

463 /* Read /proc/self/exe once so we can call addr2line later */
464 int init_symbol_handler(void) {
465 if (g_exec_path_initialized) return 1;
466 ssize_t len = readlink("/proc/self/exe", g_exec_path, sizeof(g_exec_path)-1);
467 if (len > 0 && len < (ssize_t)sizeof(g_exec_path)) {
468 g_exec_path[len] = '\0';
469 } else {
470 g_exec_path[0] = '\0';
471 }
472 g_exec_path_initialized = 1;

488 int get_caller_info(char *out, size_t outSize, unsigned int depth) {
489 if (!g_exec_path_initialized) {
490 printf("symbols handler wasn't init\n");
491 init_symbol_handler();
492 }
493
494 void *buffer[MAX_TRACE_BACK_FRAMES];
495 int nptrs = backtrace(buffer, MAX_TRACE_BACK_FRAMES);
496 if (nptrs <= (int)depth) {
497 return snprintf(out, outSize, "No stack frame at depth %u", depth) < 0 ? -1 : 0;
498 }
499
500 void *addr = buffer[depth];
501 Dl_info info;
502
503 const char *func = NULL;
504
505 if (dladdr(addr, &info) && info.dli_sname) {
506 func = info.dli_sname;
507 } else {
508 func = "[unknown]";
509 }
510
511 /* Write function name */
512 int w = snprintf(out, outSize, "Caller function: %s", func);
513 if (w < 0 || (size_t)w >= outSize) return -1;
514
515 /* Determine module path and relative address for addr2line */
516 const char *module = g_exec_path;
517 void *rel_addr = addr;
518 if (dladdr(addr, &info)) {
519 if (info.dli_fname && info.dli_fname[0] != '\0') {
520 module = info.dli_fname;
521 }
522 /* For PIE/ASLR, subtract the module base to get relative offset */
523 rel_addr = (void *)((uintptr_t)addr - (uintptr_t)info.dli_fbase);
524 }
525
526 /* If we have a module path, ask addr2line for file:line */
527 if (module[0] != '\0') {
528 char cmd[PATH_MAX_LENGTH + 100];
529 /*
530 * -f: show function name
531 * -C: demangle C++ names (harmless for C)
532 * -e: executable or shared object
533 * Using a two-stage read: first line = function, second = file:line
534 */
535 snprintf(cmd, sizeof(cmd),
536 "addr2line -f -C -e \"%s\" %p 2>/dev/null",
537 module, rel_addr);
538 FILE *fp = popen(cmd, "r");
539 if (fp) {
540 char linebuf[PATH_MAX_LENGTH + 50];
541 /* Skip the function line (we already have it from dladdr) */
542 if (fgets(linebuf, sizeof(linebuf), fp)) {
543 /* Read file:line */
544 if (fgets(linebuf, sizeof(linebuf), fp)) {
545 size_t len = strlen(linebuf);
546 if (len && linebuf[len-1] == '\n') {
547 linebuf[len-1] = '\0';
548 }
549 /* Append to output */
550 size_t cur = strlen(out);
551 snprintf(out + cur, outSize - cur, "\n at %s", linebuf);
552 }
553 }
554 pclose(fp);
555 }
556 }
557
558 return 0;
559 }
