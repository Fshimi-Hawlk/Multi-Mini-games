174void suika_updatePhysics(SuikaGame_St* game, float deltaTime) {
175 const float linearDamping = 0.99f;
176 const float angularDamping = 0.95f;
177 const float groundFriction = 0.9f;
178 const float restitution = 0.1f;
179 const float wallRestitution = 0.1f;
180 const float terminalVelocity = 800.0f;
181
182 for (int i = 0; i < SUIKA_MAX_FRUITS; i++) {
183 if (!game->fruits[i].isActive || game->fruits[i].isMerging) continue;
184
185 Fruit_St* fruit = &game->fruits[i];
186
187 fruit->velocity.y += game->gravity * deltaTime;
188
189 if (fruit->velocity.y > terminalVelocity) {
190 fruit->velocity.y = terminalVelocity;
191 }
192 if (fruit->velocity.y < -terminalVelocity) {
193 fruit->velocity.y = -terminalVelocity;
194 }
195
196 fruit->velocity.x *= linearDamping;
197 fruit->velocity.y *= linearDamping;
198 fruit->angularVelocity *= angularDamping;
199
200 fruit->position.x += fruit->velocity.x * deltaTime;
201 fruit->position.y += fruit->velocity.y * deltaTime;
202
203 fruit->rotation += fruit->angularVelocity * deltaTime;
204
205 float minX = SUIKA_CONTAINER_X + fruit->radius;
206 float maxX = SUIKA_CONTAINER_X + SUIKA_CONTAINER_WIDTH - fruit->radius;
207 float maxY = SUIKA_CONTAINER_Y + SUIKA_CONTAINER_HEIGHT - fruit->radius;
208
209 if (fruit->position.x < minX) {
210 fruit->position.x = minX;
211 fruit->velocity.x = -fruit->velocity.x * wallRestitution;
212 fruit->angularVelocity += fruit->velocity.y * 0.03f;
213 } else if (fruit->position.x > maxX) {
214 fruit->position.x = maxX;
215 fruit->velocity.x = -fruit->velocity.x * wallRestitution;
216 fruit->angularVelocity -= fruit->velocity.y * 0.03f;
217 }
218
219 if (fruit->position.y > maxY) {
220 fruit->position.y = maxY;
221 fruit->velocity.y = -fruit->velocity.y * restitution;
222 if (fabsf(fruit->velocity.y) < 20.0f) {
223 fruit->velocity.y = 0.0f;
224 }
225 fruit->velocity.x *= groundFriction;
226 fruit->angularVelocity = fruit->velocity.x * 0.02f;
227 }
228
229 for (int j = i + 1; j < SUIKA_MAX_FRUITS; j++) {
230 if (!game->fruits[j].isActive || game->fruits[j].isMerging) continue;
231
232 Fruit_St* other = &game->fruits[j];
233 float dist = Vector2Distance(fruit->position, other->position);
234 float minDist = fruit->radius + other->radius;
235
236 if (dist < minDist && dist > 0.001f) {
237 Vector2 normal = Vector2Normalize(Vector2Subtract(other->position, fruit->position));
238 float overlap = minDist - dist;
239
240 float totalMass = fruit->radius + other->radius;
241 float ratio1 = other->radius / totalMass;
242 float ratio2 = fruit->radius / totalMass;
243
244 fruit->position = Vector2Subtract(fruit->position, Vector2Scale(normal, overlap * ratio1));
245 other->position = Vector2Add(other->position, Vector2Scale(normal, overlap * ratio2));
246
247 Vector2 relativeVel = Vector2Subtract(fruit->velocity, other->velocity);
248 float velAlongNormal = Vector2DotProduct(relativeVel, normal);
249
250 if (velAlongNormal < 0) {
251 float fruitMass = fruit->radius * fruit->radius;
252 float otherMass = other->radius * other->radius;
253 float totalInvMass = 1.0f / fruitMass + 1.0f / otherMass;
254
255 float restitutionFactor = 0.5f;
256 float j = -(1.0f + restitutionFactor) * velAlongNormal / totalInvMass;
257 Vector2 impulse = Vector2Scale(normal, j);
258
259 fruit->velocity = Vector2Subtract(fruit->velocity, Vector2Scale(impulse, 1.0f / fruitMass));
260 other->velocity = Vector2Add(other->velocity, Vector2Scale(impulse, 1.0f / otherMass));
261
262 Vector2 tangent = Vector2Subtract(relativeVel, Vector2Scale(normal, velAlongNormal));
263 float tangentLen = Vector2Length(tangent);
264 if (tangentLen > 0.001f) {
265 tangent = Vector2Scale(tangent, 1.0f / tangentLen);
266 float frictionImpulse = tangentLen * 0.3f / totalInvMass;
267 fruit->velocity = Vector2Subtract(fruit->velocity, Vector2Scale(tangent, frictionImpulse / fruitMass));
268 other->velocity = Vector2Add(other->velocity, Vector2Scale(tangent, frictionImpulse / otherMass));
269 }
270
271 float angularImpulse = velAlongNormal * 0.05f;
272 fruit->angularVelocity += angularImpulse;
273 other->angularVelocity -= angularImpulse;
274 }
275 }
276 }
277 }

280void suika_checkMerging(SuikaGame_St* game) {
281 for (int i = 0; i < SUIKA_MAX_FRUITS; i++) {
282 if (!game->fruits[i].isActive || game->fruits[i].isMerging) continue;
283
284 for (int j = i + 1; j < SUIKA_MAX_FRUITS; j++) {
285 if (!game->fruits[j].isActive || game->fruits[j].isMerging) continue;
286
287 Fruit_St* f1 = &game->fruits[i];
288 Fruit_St* f2 = &game->fruits[j];
289
290 if (f1->type == f2->type) {
291 float dist = Vector2Distance(f1->position, f2->position);
292 float touchDist = f1->radius + f2->radius;
293
294 if (dist < touchDist * 1.1f) {
295 if (f1->type < FRUIT_WATERMELON) {
296 FruitType_Et newType = (FruitType_Et)(f1->type + 1);
297 const FruitProperties_St* props = suika_getFruitProperties(newType);
298
299 for (int k = 0; k < SUIKA_MAX_FRUITS; k++) {
300 if (!game->fruits[k].isActive) {
301 Vector2 midPos = Vector2Scale(Vector2Add(f1->position, f2->position), 0.5f);
302
303 game->fruits[k].position = midPos;
304 game->fruits[k].velocity = (Vector2){0.0f, 0.0f};
305 game->fruits[k].type = newType;
306 game->fruits[k].radius = props->radius;
307 game->fruits[k].rotation = (f1->rotation + f2->rotation) * 0.5f;
308 game->fruits[k].angularVelocity = (f1->angularVelocity + f2->angularVelocity) * 0.5f;
309 game->fruits[k].isActive = true;
310 game->fruits[k].isMerging = false;
311 game->fruits[k].id = game->nextFruitId++;
312
313 game->score += props->points;
314
315 f1->isActive = false;
316 f2->isActive = false;
317
318 return;
319 }
320 }
321 }
322 }
323 }
