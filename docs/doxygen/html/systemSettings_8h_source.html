163static Error_Et applyVideoSettings(VideoSettings_St settings) {
164 Error_Et err = OK;
165
166 if (settings.title != NULL) {
167 SetWindowTitle(settings.title);
168 } else {
169 log_warn("Received `title` setting was set to NULL");
170 err = ERROR_INVALID_SETTING;
171 }
172
173 if (settings.fps > 0) {
174 SetTargetFPS(settings.fps);
175 } else {
176 log_warn("Received `FPS` setting was set to zero");
177 err = ERROR_INVALID_SETTING;
178 }
179
180 if (settings.width > 0 && settings.height > 0) {
181 SetWindowSize(settings.width, settings.height);
182 } else {
183 log_warn("Received `width` and/or `height` settings was set to zero");
184 err = ERROR_INVALID_SETTING;
185 }
186
187 if (settings.fullscreen != 0 && settings.fullscreen != 1) {
188 log_warn("Received `fullscreen` setting exceed value range (0 or 1)");
189 err = ERROR_INVALID_SETTING;
190 settings.fullscreen = DEFAULT_VIDEO_SETTING_FULLSCREEN;
191 }
192
193 if (settings.vsync != 0 && settings.vsync != 1) {
194 log_warn("Received `vsync` setting exceed value range (0 or 1)");
195 err = ERROR_INVALID_SETTING;
196 settings.vsync = DEFAULT_VIDEO_SETTING_VSYNC;
197 }
198
199 if (settings.borderless != 0 && settings.borderless != 1) {
200 log_warn("Received `borderless` setting exceed value range (0 or 1)");
201 err = ERROR_INVALID_SETTING;
202 settings.borderless = DEFAULT_VIDEO_SETTING_BORDERLESS;
203 }
204
205 if (settings.resizable != 0 && settings.resizable != 1) {
206 log_warn("Received `resizable` setting exceed value range (0 or 1)");
207 err = ERROR_INVALID_SETTING;
208 settings.resizable = DEFAULT_VIDEO_SETTING_RESIZABLE;
209 }
210
211 SetWindowState(
212 (settings.fullscreen ? FLAG_FULLSCREEN_MODE : 0)
213 | (settings.vsync ? FLAG_VSYNC_HINT : 0)
214 | (settings.borderless ? FLAG_WINDOW_UNDECORATED : 0)
215 | (settings.resizable ? FLAG_WINDOW_RESIZABLE : 0)
216 );
217
218 return err;
219}
